CREATE TABLE `juego`.`numerosj` ( `idnj` INT(10) NOT NULL AUTO_INCREMENT , `b` INT(5) NOT NULL , `i` INT(5) NOT NULL , 
`n` INT(5) NOT NULL , `g` INT(5) NOT NULL , `o` INT(5) NOT NULL , `idj` VARCHAR(200) NOT NULL , PRIMARY KEY (`idnj`)) ENGINE = InnoDB; 

ALTER TABLE `juego`.`numerosj` ADD FOREIGN KEY (`idj`) REFERENCES `jugador`(`idj`) ON DELETE RESTRICT ON UPDATE RESTRICT;
